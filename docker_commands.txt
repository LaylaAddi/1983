# Start services
docker compose up -d

# Stop services
docker compose down

# Restart after code changes
docker compose restart web

# Full rebuild (after Dockerfile/requirements.txt changes)
docker compose down
docker compose build --no-cache
docker compose up -d

# View logs
docker compose logs -f web
docker compose logs --tail=100 web

# Django shell
docker compose exec web python manage.py shell

# Run migrations
docker compose exec web python manage.py makemigrations
docker compose exec web python manage.py migrate

# Create superuser
docker compose exec web python manage.py createsuperuser

# Collect static files
docker compose exec web python manage.py collectstatic --noinput
Migration Workflow We Used
powershell# 1. Made model changes in accounts/models.py
# 2. Created migration
docker compose exec web python manage.py makemigrations accounts

# Expected output:
# Migrations for 'accounts':
#   accounts/migrations/0004_referralsettings.py
#     - Create model ReferralSettings

# 3. Applied migration
docker compose exec web python manage.py migrate accounts

# Expected output:
# Running migrations:
#   Applying accounts.0004_referralsettings... OK

# 4. Verified in shell
docker compose exec web python manage.py shell
